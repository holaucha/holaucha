MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK' OR 'EXIT' OR 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN 'GET_URL'.
      CASE marked.
        WHEN gv_rb01.
          gv_selid = 'AA'.
        WHEN gv_rb02.
          gv_selid = 'LH'.
        WHEN gv_rb03.
          gv_selid = 'DL'.
      ENDCASE.
      PERFORM get_url.
      gv_carrid = gs_scarr-carrid.
      gv_url = gs_scarr-url.
    WHEN 'SEL'.
      PERFORM get_spfli.
      CALL SCREEN 200.

  ENDCASE.

ENDMODULE.
